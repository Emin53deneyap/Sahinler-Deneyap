#include <Arduino.h>
#include "v1a_camera.h"

// Pin tanımlamaları - Deneyap Kart'ınızdaki bağlantılarınıza göre kontrol edin
const int gazPin = A0;      // Gaz sensörünün bağlı olduğu analog pin
const int rolePin = D0;     // Su pompası (röle) pini

void setup() {
  // Processing ile düzgün haberleşme için 115200 hızını kullanıyoruz
  Serial.begin(115200); 
  
  pinMode(rolePin, OUTPUT);
  digitalWrite(rolePin, LOW); // Başlangıçta pompa kapalı
  
  // Kamera başlatma kontrolü
  if (camera_init() != 0) {
    while (1); // Kamera başlamazsa burada takılı kalır, hatayı fark edersiniz
  }
  camera_config(2, 2); // 320x240 çözünürlük
}

void loop() {
  int hamGaz = analogRead(gazPin);
  int gazSeviyesi = 0;

  // Gaz seviyesi belirleme mantığı
  if (hamGaz < 200) {
    gazSeviyesi = 0; // Güvenli
  } else if (hamGaz >= 200 && hamGaz < 400) {
    gazSeviyesi = 1; // Hafif gaz
  } else if (hamGaz >= 400 && hamGaz < 600) {
    gazSeviyesi = 2; // Şüpheli
  } else {
    gazSeviyesi = 3; // YANGIN/TEHLİKE!
  }

  // Processing programına seviyeyi gönder (Arayüzde güncelleme yapmak için)
  Serial.print("SEVIYE:");
  Serial.println(gazSeviyesi);

  // POMPA KONTROLÜ: Sadece kritik seviyede (3) çalışır
  if (gazSeviyesi == 3) {
    digitalWrite(rolePin, HIGH); 
  } else {
    digitalWrite(rolePin, LOW); 
  }

  // FOTOĞRAF KONTROLÜ: Seviye 0'dan büyükse fotoğraf gönder
  if (gazSeviyesi > 0) {
    camera_frame_t *frame = camera_get_frame();
    if (frame) {
      Serial.println("START_IMAGE"); 
      Serial.write(frame->buf, frame->len);
      Serial.println("END_IMAGE");
      camera_return_frame(frame);
    }
    delay(3000); // 3 saniyede bir fotoğraf gönderir
  } else {
    delay(500); // Normal durumda yarım saniyede bir kontrol eder
  }
}
